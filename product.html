public class Product {

    private String _name;
    private String _brand;
    private double _price;
    private int _stock;


    public Product(String name, String brand, double price, int stock) {
        _name = name;
        _brand = brand;
        _price = price;
        _stock = stock;

    }

    public String getName() {
        return _name;
    }

    public String getBrand() {
        return _brand;
    }

    public double getPrice() {
        return _price;
    }

    public int getStock() {
        return _stock;
    }


    public String getDescription() {
        if (_stock != 0) {
            return  _name + " MARKA PRODUKTU : " + _brand+ " CENA: "   + String.format("%.2f", _price)+ " zl" + " ILOSC: " + _stock + " WARTOSC TWOICH PRODUKTOW: " + TotalWorth() + " zl" ;
        } else {
            return _name + " MARKA PRODUKTU: " + _brand+ "CENA: "  +  String.format("%.2f", _price) + " zl";
        }
    }

    public double TotalWorth() {

        double HowMuch = _price * _stock;
        return HowMuch;
    }
}
