            public class BubbleSort {
                public static void Sort(MyIntCollection collection, Counter loopCounter, Counter comparisonCounter) {
                    int size = collection.length();
            
                    // loop to access each array element
                    for (int i = 0; i < size - 1; i++) {
                        loopCounter.PlusOne();
            
                        // loop to compare array elements
                        for (int j = 0; j < size - i - 1; j++) {
                            loopCounter.PlusOne();
            
            
                            // compare two adjacent elements
                            // change > to < to sort in descending order
                            comparisonCounter.PlusOne();
                            if (collection.getIntAt(j) > collection.getIntAt(j +1)) {
            
                                // swapping occurs if elements
                                // are not in the intended order
                                int temporaryValue = collection.getIntAt(j);
                                collection.setIntAt(j, collection.getIntAt(j+1));
                                collection.setIntAt(j + 1, temporaryValue);
                            }
                        }
            
                    }
            
                }
            }
