public static void DisplayPath(Node[] currentPath ){
    if ( currentPath != null){
        System.out.println("We found your connection");
        System.out.println("Your path: ");
        Node last = currentPath[currentPath.length - 1];
        for (Node steps : currentPath) {

            System.out.print(steps);
            if( last != steps) {
            System.out.print("=>");
            }
        }
        System.out.println();
    } else {
        System.out.println("There is no connection");
    }
}
